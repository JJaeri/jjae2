function generateCalendar() { const calendar = document.getElementById('calendar'); const today = new Date(); const year = today.getFullYear(); const month = today.getMonth(); // 달력 헤더 생성 const daysOfWeek = ['일', '월', '화', '수', '목', '금', '토']; let headerRow = ''; for (let day of daysOfWeek) { headerRow += `${day}`; } headerRow += ''; calendar.innerHTML = headerRow; // 달력 날짜 생성 const firstDayOfMonth = new Date(year, month, 1).getDay(); const daysInMonth = new Date(year, month + 1, 0).getDate(); let date = 1; for (let i = 0; i < 6; i++) { let row = ''; for (let j = 0; j < 7; j++) { if (i === 0 && j < firstDayOfMonth) { row += ''; } else if (date > daysInMonth) { break; } else { row += `${date}`; date++; } } row += ''; calendar.innerHTML += row; } }
